INITIAL CSS

.div1 {
  margin: 1px;
}

.div1 .div2 {
  margin: 1px 2px;
}

DESIRED TREE
{
    ".div1" : {
        ".div2" : {
            "properties" : [
                "margin: 1px 2px;"
            ]
        },
        "properties" : [
            "margin: 1px;"
        ]
    }
}

HTML OUTPUT FROM TREE
<div class = 'selector'>.div1 {
    margin:<span class = 'margin 1px px'>1px</span>;<br>

    <div class = 'selector'>.div2 {
        margin:<span class = 'margin 1px px'>1px</span><span class = 'margin 2px px'>2px</span>;<br>
    }</div>
}</div>
